macro_rules! jit_struct(
    ($($name:ident: $ty:ty),*) => ({
        let structure = Type::create_struct([
            $(get::<$ty>()),*
        ].as_mut_slice());
        structure.set_names(&[$(stringify!($name)),*]);
        structure
    });
    ($($ty:ty),+ ) => ({
        Type::create_struct([
            $(get::<$ty>()),+
        ].as_mut_slice())
    })
);
